public function Records_DrawRecords
	Object.Minutes = saveRAM[arrayPos0]
	Object.Minutes /= 6000

	Object.Seconds = saveRAM[arrayPos0]
	Object.Seconds /= 100
	Object.Seconds %= 60

	Object.MilliSeconds = saveRAM[arrayPos0]
	Object.MilliSeconds %= 100

	temp0 = object.xpos
	temp0 >>= 16
	temp0 -= 16
	DrawNumbers(26, temp0, temp1, Object.Minutes, 2, 8, 1)
	temp0 += 24
	DrawNumbers(26, temp0, temp1, Object.Seconds, 2, 8, 1)
	temp0 += 24
	DrawNumbers(26, temp0, temp1, Object.MilliSeconds, 2, 8, 1)
end function


event ObjectUpdate
	object.ypos = Object.Record
	object.ypos *= 192
	object.ypos -= MenuControl.SMYPos
	if object.ypos < -384
		object.ypos += 576
	end if
	object.ypos += 224
	object.ypos <<= 16
end event


event ObjectDraw
	DrawSprite(47)
	switch object.propertyValue
	case FIRST_RECORD
		temp0 = LargeIcon.Portrait[-1]
		temp0 += 36
		DrawSprite(temp0)
		DrawSprite(44)
		break

	case SECOND_RECORD
		temp0 = LargeIcon.Portrait[-2]
		temp0 += 36
		DrawSprite(temp0)
		DrawSprite(45)
		break

	case THIRD_RECORD
		temp0 = LargeIcon.Portrait[-3]
		temp0 += 36
		DrawSprite(temp0)
		DrawSprite(46)
		break
	end switch

	arrayPos0 = temp0
	arrayPos0 -= 36
	arrayPos0 *= 18

	temp0 = Object.Record
	temp0 *= 6

	arrayPos0 += temp0
	arrayPos0 += 48

	temp1 = object.ypos
	temp1 >>= 16
	temp1 += 97
	CallFunction(Records_DrawRecords)

	temp1 += 24
	arrayPos0 += 2
	CallFunction(Records_DrawRecords)

	temp1 += 24
	arrayPos0 += 2
	CallFunction(Records_DrawRecords)

#platform: Standard
	switch MenuControl.State
	case MENUCONTROL_SM_CONTROLS
		switch MenuControl.SMCurrentButton
		case 0
			DrawSprite(48)
			break
		case 1
			DrawSprite(49)
			break
		case 2
			DrawSprite(50)
			break
		end switch
		break

	case MENUCONTROL_SM_CONFIRMED
		temp0 = MenuControl.Timer
		temp0 &= 3
		temp0 >>= 1
		if temp0 == 0
			DrawSprite(50)
		end if
		break

	case MENUCONTROL_SM_LOAD_LEVEL
		DrawSprite(50)
		break

	case MENUCONTROL_SM_COMFIRMED_LEAVE
		temp0 = MenuControl.Timer
		temp0 &= 3
		temp0 >>= 1
		if temp0 == 0
			DrawSprite(48)
		end if
		break

	end switch
#endplatform

#platform: Mobile
	DrawSprite(51) // Blocks the PC buttons
#endplatform
end event
